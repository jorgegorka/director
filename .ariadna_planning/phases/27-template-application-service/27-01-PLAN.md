Create `app/services/apply_role_template_service.rb` following the project's service pattern:

```ruby
class ApplyRoleTemplateService
  Result = Data.define(:created, :skipped, :errors, :created_roles) do
    def created_count = created
    def skipped_count = skipped
    def success? = errors.empty?
    def total = created + skipped
    def summary
      parts = []
      parts << "Created #{created} role#{'s' unless created == 1}" if created > 0
      parts << "Skipped #{skipped} existing" if skipped > 0
      parts << "#{errors.size} error#{'s' unless errors.size == 1}" if errors.any?
      parts.join(", ")
    end
  end

  attr_reader :company, :template_key, :parent_role

  def initialize(company:, template_key:, parent_role: nil)
    @company = company
    @template_key = template_key
    @parent_role = parent_role
  end

  def self.call(**kwargs)
    new(**kwargs).call
  end

  def call
    template = RoleTemplateRegistry.find(template_key)
    created = 0
    skipped = 0
    errors = []
    created_roles = []
    roles_by_title = {}  # tracks created roles for parent lookups

    template.roles.each do |template_role|
      existing = company.roles.find_by(title: template_role.title)
      if existing
        skipped += 1
        roles_by_title[template_role.title] = existing
        next
      end

      parent = resolve_parent(template_role, roles_by_title)
      role = company.roles.build(
        title: template_role.title,
        description: template_role.description,
        job_spec: template_role.job_spec,
        parent: parent
      )

      if role.save
        created += 1
        created_roles << role
        roles_by_title[template_role.title] = role
        assign_skills(role, template_role.skill_keys)
      else
        errors << "#{template_role.title}: #{role.errors.full_messages.join(', ')}"
      end
    end

    Result.new(created: created, skipped: skipped, errors: errors.freeze, created_roles: created_roles.freeze)
  end

  private

  def resolve_parent(template_role, roles_by_title)
    if template_role.parent.nil?
      parent_role  # the caller-supplied parent (for Apply All nesting under CEO)
    else
      roles_by_title[template_role.parent]  # parent from this template run or pre-existing
    end
  end

  def assign_skills(role, skill_keys)
    return if skill_keys.empty?
    skills = company.skills.where(key: skill_keys)
    skills.each do |skill|
      role.role_skills.create(skill: skill)
    rescue ActiveRecord::RecordInvalid
      # skip duplicates silently (RoleSkill validates uniqueness)
    end
  end
end
```

Key design decisions:
- `Result` uses `Data.define` (consistent with Phase 26's TemplateRole/Template pattern)
- `parent_role:` optional parameter allows Apply All to nest department roots under CEO
- `resolve_parent` checks `roles_by_title` hash which includes BOTH newly-created AND pre-existing (skipped) roles -- this ensures children of skipped roles still get correct parents
- `assign_skills` uses `company.skills.where(key:)` -- NEVER crosses tenant boundaries (locked decision)
- NO transaction wrapper (locked decision: partial success preferred)
- `role.save` (not `save!`) -- errors are collected into the result, not raised
- `assign_skills` uses `create` with rescue for duplicates (idempotent)
- Title uniqueness check via `company.roles.find_by(title:)` leverages SQLite COLLATE NOCASE on the roles table (locked decision)
