Create the Hookable concern that detects task status transitions and automatically enqueues hook execution jobs. This plan establishes the trigger detection and enqueueing pipeline; the actual hook dispatch logic (trigger_agent, webhook) is implemented in plan 19-02.

    ```ruby
    module Hookable
      extend ActiveSupport::Concern

      private

      # Map task status values to lifecycle event strings
      HOOKABLE_TRANSITIONS = {
        "in_progress" => AgentHook::AFTER_TASK_START,
        "completed" => AgentHook::AFTER_TASK_COMPLETE
      }.freeze

      def enqueue_hooks_for_transition
        return unless saved_change_to_status?
        return unless assignee_id.present?

        lifecycle_event = HOOKABLE_TRANSITIONS[status]
        return unless lifecycle_event

        hooks = assignee.agent_hooks.enabled.for_event(lifecycle_event).ordered
        hooks.each do |hook|
          execution = HookExecution.create!(
            agent_hook: hook,
            task: self,
            company_id: company_id,
            status: :queued,
            input_payload: build_hook_input_payload(hook)
          )
          ExecuteHookJob.perform_later(execution.id)
        end
      end

      def build_hook_input_payload(hook)
        {
          task_id: id,
          task_title: title,
          task_status: status,
          agent_id: assignee_id,
          agent_name: assignee.name,
          lifecycle_event: hook.lifecycle_event,
          hook_name: hook.name,
          action_type: hook.action_type,
          action_config: hook.action_config,
          triggered_at: Time.current.iso8601
        }
      end
    end
    ```

    Key design decisions:
    - Uses `saved_change_to_status?` (after_commit context) to detect transitions — consistent with how Task already uses `saved_change_to_assignee_id?` in trigger_assignment_wake
    - HOOKABLE_TRANSITIONS maps the two triggering statuses to lifecycle event strings — open/blocked/cancelled transitions are ignored
    - Creates HookExecution with :queued status and input_payload capturing the full context before enqueueing the job
    - Each hook gets its own HookExecution and job — they execute independently

    - The on: [:create, :update] is needed because a task could be created directly in in_progress/completed status

    Create `app/jobs/execute_hook_job.rb`:

    ```ruby
    class ExecuteHookJob < ApplicationJob
      queue_as :default

      retry_on StandardError, wait: :polynomially_longer, attempts: 3
      discard_on ActiveJob::DeserializationError

      def perform(hook_execution_id)
        execution = HookExecution.find_by(id: hook_execution_id)
        return unless execution
        return if execution.completed? || execution.failed?

        ExecuteHookService.call(execution)
      end
    end
    ```

    Key design decisions:
    - `retry_on StandardError, wait: :polynomially_longer, attempts: 3` — requirement ACT-03 says polynomial backoff with 3 attempts. Rails' built-in `:polynomially_longer` uses `(executions**4) + (Kernel.rand * executions * jitter)` which satisfies "polynomial backoff"
    - `discard_on ActiveJob::DeserializationError` — safely handles deleted records
    - Guard clause: skip if execution is already completed or failed (idempotency for retries)
    - Receives `hook_execution_id` (integer) not the record itself — avoids serialization issues and allows the job to handle missing records gracefully
    - Delegates entirely to `ExecuteHookService.call(execution)` which will be implemented in plan 19-02. For now, calling a non-existent service is fine because the job tests will stub the service call

    NOTE: Do NOT create ExecuteHookService yet — it is plan 19-02's responsibility. The job tests will use stubs/mocks for the service.

    ```ruby
    class HookableTest < ActiveSupport::TestCase
      setup do
        @company = companies(:acme)
        @claude_agent = agents(:claude_agent)

        @task = tasks(:design_homepage)  # in_progress, assigned to claude_agent

      test "transitioning task to completed enqueues ExecuteHookJob for matching enabled hooks" do
        # design_homepage is in_progress, assigned to claude_agent
        # claude_agent has claude_validation_hook (after_task_complete, enabled)
        assert_difference "HookExecution.count", 1 do
          assert_enqueued_with(job: ExecuteHookJob) do
            @task.update!(status: :completed)
          end
        end

        execution = HookExecution.last
        assert_equal "queued", execution.status
        assert_equal @validation_hook.id, execution.agent_hook_id
        assert_equal @task.id, execution.task_id
        assert_equal @company.id, execution.company_id
      end

      test "input_payload contains task and hook context" do
        @task.update!(status: :completed)
        execution = HookExecution.last

        payload = execution.input_payload
        assert_equal @task.id, payload["task_id"]
        assert_equal @task.title, payload["task_title"]
        assert_equal "completed", payload["task_status"]
        assert_equal @claude_agent.id, payload["agent_id"]
        assert_equal @claude_agent.name, payload["agent_name"]
        assert_equal "after_task_complete", payload["lifecycle_event"]
        assert_equal "trigger_agent", payload["action_type"]
        assert payload["triggered_at"].present?
    ```
